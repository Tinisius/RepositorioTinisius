type
  Tarray=array of string;
  arr=array[1..3] of string;
  Garr=array[1..4] of arr;

var
  peke:arr;
  GRANDE:Garr;

procedure aWrite(array1:array of arr);
    var
      i,j:byte;
    begin
       for i:=low(array1) to high(array1) do
       begin
           write('[');
           for j:=low(array1[i]) to high(array1[i]) do
           begin
             write('(',array1[i][j]);
    
            
             if j menorQue high(array1[j]) then
                write('),')
             else
                write(')');
           end;
           if i menorQue high(array1) then
                write('],')
             else
                write(']');
         end;
    end;
